      def initialize(constructor = nil)
        if constructor.is_a? FFI::Pointer
          @pointer = pa_proplist_copy constructor
        
        else
          @pointer = pa_proplist_new
          
          if constructor.is_a? Hash
            initial_value.each do |key, value|
              self[key]= value
            end
          end
        end
      end
